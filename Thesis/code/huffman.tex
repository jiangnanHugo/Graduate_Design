 def build_huffman(frequenties):
     Q=Queue.PriorityQueue() #初始化优先队列 Q;
     for v in frequenties:  #((freq,word),index)
         Q.put(v)
     idx=0
     while Q.qsize()>1:#由底至顶合并节点
         l,r=Q.get(),Q.get()
         node=Node(l,r,idx)
         idx+=1
         freq=l[0]+r[0]
         Q.put((freq,node))
     return Q.get()[1] # 霍夫曼树
\end{minted}
